## [COTE] 문자여열 압축

```javascript
function solution(s) {
    if(s.length === 1) return 1;

    // 문자열 길이 num 만큼 압축하기
    function makingCompress(num){
        let ansString='';
        for(let idx=0; idx<s.length ; ){
            let ch = s.substr(idx,num);
            // idx부터 num개수 만큼 자르기
            let cnt=1;
            for(let i=idx+num; i<s.length; i+=num){
                if(ch===s.substr(i,num)){
                    cnt++;
                }
                else break;
            }
            // cnt 개수 1보다 크면 문자열 압축하기
            if(cnt > 1){
                ansString += cnt.toString() + ch;
            }
            else ansString += ch;

            idx+=cnt*num;
            // 큰 for문 조건
        }
        return ansString.length;
    }
    let n=1;
    let answer = 1000;
    while(n<s.length){
        let value = makingCompress(n);
        answer = Math.min(answer,value);
        n++;
    }
    return answer;
}

const str = "ababcdcdababcdcd";
console.log('정답=' +solution(str));
```
