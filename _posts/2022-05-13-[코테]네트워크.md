## [STUDY] 네트워크

```javascript
function solution(n, computers){
    const check = new Array(n).fill(false);
    let answer =0;

    function dfs(target){
        check[target]=true;        
        for(let i=0; i<computers[target].length; i++){
            if(target===i) continue;
            if(computers[target][i] && !check){
                dfs(i)
            }
        }
    }
    for(let i=0; i<n; i++){
        if(!check[i]){
            answer++;
            dfs(i);
        }
        if(check.every((el)=>el)) break;
    }
    return answer;
}
```
